## Implementation

```text
AiPrRateFunction.java:

  @ApplicationScoped
  public class AiPrRateFunction implements MetricFunction {

    @Inject
    PullRequestQuery prQuery;

    @Override
    public String getMetricName() {
      return "ai_assisted_pr_rate";
    }

    @Override
    public MetricResult compute(UUID teamId, Instant windowStart) {
      // 1. Count AI-assisted PRs (declared + inferred)
      long aiCount = prQuery.countByTeam(teamId, windowStart,
        "ai_assisted = true OR ai_usage_inferred = true");

      // 2. Count classified PRs (non-null ai_assisted OR ai_usage_inferred is set)
      long classifiedCount = prQuery.countByTeam(teamId, windowStart,
        "ai_assisted IS NOT NULL OR ai_usage_inferred IS NOT NULL");

      // 3. Compute rate
      double rate = classifiedCount == 0 ? 0.0 : (double) aiCount / classifiedCount;

      // 4. Round to 4 decimal places
      rate = Math.round(rate * 10000.0) / 10000.0;

      return new MetricResult(rate, null, null);
    }
  }
```

## SQL (alternative: use database functions)

```sql
-- If you prefer SQL-based metrics:
CREATE OR REPLACE FUNCTION fn_ai_pr_rate(p_team_id UUID, p_window_start TIMESTAMPTZ)
RETURNS TABLE(metric_value NUMERIC, confidence_score INT, confidence_issue TEXT) AS $$
DECLARE
  v_ai_count BIGINT;
  v_classified_count BIGINT;
BEGIN
  SELECT
    COUNT(*) FILTER (WHERE ai_assisted = true OR ai_usage_inferred = true),
    COUNT(*) FILTER (WHERE ai_assisted IS NOT NULL OR ai_usage_inferred IS NOT NULL)
  INTO v_ai_count, v_classified_count
  FROM pull_requests pr
  JOIN repositories r ON r.id = pr.repository_id
  WHERE r.team_id = p_team_id
    AND pr.created_at_source >= p_window_start;

  RETURN QUERY SELECT
    CASE WHEN v_classified_count = 0 THEN 0::NUMERIC
         ELSE ROUND(v_ai_count::NUMERIC / v_classified_count, 4)
    END,
    NULL::INT,
    NULL::TEXT;
END;
$$ LANGUAGE plpgsql;
```
