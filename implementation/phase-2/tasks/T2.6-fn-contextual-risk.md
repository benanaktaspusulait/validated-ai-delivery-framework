DEVELOPER_EXPERIENCE (W_exp = 0.05):

  Low (internal tooling):  0
  Medium (standard product): 1
  High (payment, auth, security, regulated): 3

CHANGED_LINES (W_size = 0.15):
  < 50:    0
  50-150:  1
  150-300: 2

SECURITY_SENSITIVITY (W_security = 0.15):
  None:                          0
  Infrastructure/infra paths:    1

CROSS_TEAM_IMPACT (W_cross = 0.10):
  No cross-team files:           0
  Shared library or module:      1
  Cross-service contract:        2

REVIEWER_LOAD (W_reviewer = 0.05):
  Light (< 3 open PRs/reviewer): 0
  Normal (3-5):                   1
  Heavy (> 5):                    2

CODEBASE_FAMILIARITY (W_familiarity = 0.10):
  Many local examples (> 20 commits in path area, 6 months):  0
  Some comparable patterns (5-20 commits):                     1
  Few or no comparable examples (< 5 commits):                 2

CHANGE_FRESHNESS (W_freshness = 0.05):
  Stable area (no recent changes):  0
  Modified within 30 days:          1
  High churn (> 10 modifications in 3 months): 2

OWNERSHIP_BOUNDARY (W_boundary = 0.05):
  Same team:                        0
  Shared module:                    1
  Multiple teams or service owners: 2

  @ApplicationScoped
  public class RiskConfig {
    // Weights (must sum to 1.0)
    public static final double W_TASK = 0.15;
    public static final double W_EXP = 0.05;
    public static final double W_DOMAIN = 0.15;
    public static final double W_SIZE = 0.15;
    public static final double W_SECURITY = 0.15;
    public static final double W_CROSS = 0.10;
    public static final double W_REVIEWER = 0.05;
    public static final double W_FAMILIARITY = 0.10;
    public static final double W_FRESHNESS = 0.05;
    public static final double W_BOUNDARY = 0.05;

    // Sensitive path patterns
    public static final List<String> SECURITY_SENSITIVE_PATHS = List.of(
      "**/auth/**", "**/payment/**", "**/security/**", "**/terraform/**"
    );
    public static final List<String> INFRA_PATHS = List.of(
      "**/infra/**", "**/infrastructure/**", "**/deploy/**"
    );
    public static final List<String> CROSS_TEAM_PATHS = List.of(
      "openapi/**", "proto/**", "shared/**"
    );

    // Thresholds
    public static final int HIGH_RISK_THRESHOLD = 10;
    public static final int REVIEWER_LOAD_LIGHT = 3;
    public static final int REVIEWER_LOAD_HEAVY = 5;
    public static final int FAMILIARITY_MANY = 20;
    public static final int FAMILIARITY_SOME = 5;
    public static final int FRESHNESS_RECENT_DAYS = 30;
    public static final int FRESHNESS_CHURN_COUNT = 10;
  }

RiskFactorScorer.java:

  @ApplicationScoped
  public class RiskFactorScorer {

    @Inject
    RiskDataQuery riskDataQuery;

    public int scoreTaskType(String taskType) {
      return switch (taskType.toLowerCase()) {
        case "documentation" -> 0;
        case "test generation" -> 1;
        case "internal tooling" -> 1;
        case "business logic" -> 2;
        case "payment", "auth", "security" -> 3;
        case "cross-service contract" -> 3;
        default -> 1;  // default to medium
      };
    }

    public int scoreDeveloperExperience(String seniority) {
      return switch (seniority.toLowerCase()) {
        case "senior" -> 0;
        case "mid", "mid-level" -> 1;
        case "junior" -> 2;
        default -> 1;
      };
    }

    public int scoreDomainCriticality(String criticality) {
      return switch (criticality.toLowerCase()) {
        case "low" -> 0;
        case "medium" -> 1;
        case "high" -> 3;
        default -> 1;
      };
    }

    public int scoreChangedLines(int lines) {
      if (lines < 50) return 0;
      if (lines < 150) return 1;
      if (lines < 300) return 2;
      return 3;
    }

    public int scoreSecuritySensitivity(List<String> changedFiles) {
      for (String file : changedFiles) {
        for (String pattern : RiskConfig.SECURITY_SENSITIVE_PATHS) {
          if (matchesGlob(pattern, file)) return 3;
        }
        for (String pattern : RiskConfig.INFRA_PATHS) {
          if (matchesGlob(pattern, file)) return 1;
        }
      }
      return 0;
    }

    public int scoreCrossTeamImpact(List<String> changedFiles) {
      for (String file : changedFiles) {
        for (String pattern : RiskConfig.CROSS_TEAM_PATHS) {
          if (matchesGlob(pattern, file)) return 2;
        }
      }
      return 0;  // simplified: check if files cross repo boundaries
    }

    public int scoreReviewerLoad(int openPrsPerReviewer) {
      if (openPrsPerReviewer < RiskConfig.REVIEWER_LOAD_LIGHT) return 0;
      if (openPrsPerReviewer <= RiskConfig.REVIEWER_LOAD_HEAVY) return 1;
      return 2;
    }

    public int scoreCodebaseFamiliarity(UUID repositoryId, List<String> changedFiles) {
      // Count commits in similar path areas over last 6 months
      int commitCount = riskDataQuery.countCommitsInPathArea(repositoryId, changedFiles, 180);
      if (commitCount >= RiskConfig.FAMILIARITY_MANY) return 0;
      if (commitCount >= RiskConfig.FAMILIARITY_SOME) return 1;
      return 2;
    }

    public int scoreChangeFreshness(UUID repositoryId, List<String> changedFiles) {
      // Check if files were modified recently
      boolean modifiedRecently = riskDataQuery.wereFilesModifiedWithinDays(repositoryId, changedFiles, RiskConfig.FRESHNESS_RECENT_DAYS);
      boolean highChurn = riskDataQuery.hasHighChurnHistory(repositoryId, changedFiles, RiskConfig.FRESHNESS_CHURN_COUNT, 90);

      if (highChurn) return 2;
      if (modifiedRecently) return 1;
      return 0;
    }

    public int scoreOwnershipBoundary(UUID repositoryId, List<String> changedFiles, UUID teamId) {
      // Check if files are owned by multiple teams
      int teamCount = riskDataQuery.countTeamsWithCommitsInFiles(repositoryId, changedFiles, 180);
      if (teamCount > 1) return 2;
      if (teamCount > 0) return 0;
      return 1;  // unknown ownership
    }

    private boolean matchesGlob(String pattern, String path) {
      // Simple glob matching: convert ** to regex
      String regex = pattern.replace("**/", ".*(/|^)").replace("/", "\\/");
      return path.matches(regex);
    }
  }

ContextualRiskFunction.java:

  @ApplicationScoped
  public class ContextualRiskFunction {

    @Inject
    RiskFactorScorer scorer;
    @Inject
    RiskDataQuery dataQuery;

    public RiskResult compute(UUID prId) {
      // 1. Load PR and repository data
      PullRequest pr = dataQuery.findPrWithRepo(prId);
      if (pr == null) return RiskResult.empty();

      // 2. Score all 10 factors
      int taskType = scorer.scoreTaskType(determineTaskType(pr));
      int devExperience = scorer.scoreDeveloperExperience(pr.getAuthorSeniority());
      int domain = scorer.scoreDomainCriticality(pr.getRepository().getDomainCriticality());
      int size = scorer.scoreChangedLines(pr.getChangedLines());
      int security = scorer.scoreSecuritySensitivity(pr.getChangedFiles());
      int crossTeam = scorer.scoreCrossTeamImpact(pr.getChangedFiles());
      int reviewerLoad = scorer.scoreReviewerLoad(dataQuery.countOpenPrsPerReviewer(pr.getRepositoryId()));
      int familiarity = scorer.scoreCodebaseFamiliarity(pr.getRepositoryId(), pr.getChangedFiles());
      int freshness = scorer.scoreChangeFreshness(pr.getRepositoryId(), pr.getChangedFiles());
      int boundary = scorer.scoreOwnershipBoundary(pr.getRepositoryId(), pr.getChangedFiles(), pr.getTeamId());

      // 3. Compute weighted score
      double rawScore =
        taskType * RiskConfig.W_TASK +
        devExperience * RiskConfig.W_EXP +
        domain * RiskConfig.W_DOMAIN +
        size * RiskConfig.W_SIZE +
        security * RiskConfig.W_SECURITY +
        crossTeam * RiskConfig.W_CROSS +
        reviewerLoad * RiskConfig.W_REVIEWER +
        familiarity * RiskConfig.W_FAMILIARITY +
        freshness * RiskConfig.W_FRESHNESS +
        boundary * RiskConfig.W_BOUNDARY;

      // Scale to 0-15 range (max possible raw score = 5 * 0.15 + 3 * 0.05 + ... = ~2.5)
      int riskScore = (int) Math.round(rawScore * 6);  // 2.5 * 6 = 15

      // 4. Collect signals (factors with score >= 2)
      List<String> signals = new ArrayList<>();
      if (taskType >= 2) signals.add("high_risk_task_type");
      if (size >= 2) signals.add("large_pr");
      if (security >= 2) signals.add("security_sensitive_path");
      if (crossTeam >= 2) signals.add("cross_team_impact");
      if (familiarity >= 2) signals.add("low_familiarity");
      if (freshness >= 2) signals.add("high_churn_area");
      if (boundary >= 2) signals.add("ownership_boundary");

      // 5. Determine required reviewers
      List<String> requiredReviewers = new ArrayList<>();
      if (riskScore >= RiskConfig.HIGH_RISK_THRESHOLD) {
        requiredReviewers.add("senior_engineer");
        requiredReviewers.add("code_owner");
      }
      if (security >= 2) {
        requiredReviewers.add("appsec");
      }

      return new RiskResult(riskScore, signals, requiredReviewers);
    }

    private String determineTaskType(PullRequest pr) {
      // Heuristic: determine task type from title, labels, or file patterns
      String title = pr.getTitle().toLowerCase();
      if (title.contains("fix") || title.contains("bug")) return "business logic";
      if (title.contains("test")) return "test generation";
      if (title.contains("doc") || title.contains("readme")) return "documentation";
      if (title.contains("refactor")) return "internal tooling";
      // Default: check file patterns
      return "business logic";
    }
  }
