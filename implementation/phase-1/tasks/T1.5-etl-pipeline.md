## Objective

Build ETL pipeline that transforms `raw_events` into normalized tables (`pull_requests`, `review_analytics`, `jira_issues`).

Step 1: Create EtlOrchestrator.java

  @Scheduled(every = "15m")  // or cron: "0 */15 * * * *"
  public void runEtl() {
    // 1. Fetch pending events
    List<RawEvent> events = rawEventRepository.findPending(limit: 1000);

    for (RawEvent event : events) {
      try {
        // 2. Mark as processing
        rawEventStatusUpdater.updateStatus(event.getId(), ProcessingStatus.PROCESSING);

        // 3. Dispatch to correct processor
        NormalizationResult result = eventDispatcher.dispatch(event);

        // 4. Mark as completed or failed
        if (result.isSuccess()) {
          rawEventStatusUpdater.updateStatus(event.getId(), ProcessingStatus.COMPLETED);
        } else {
          handleFailure(event, result.getError());
        }
      } catch (Exception e) {
        handleFailure(event, e.getMessage());
      }
    }
  }

  private void handleFailure(RawEvent event, String error) {
    int retryCount = event.getRetryCount() + 1;
    if (retryCount >= 3) {
      rawEventStatusUpdater.updateStatus(event.getId(), ProcessingStatus.DEAD_LETTER, error);
    } else {
      rawEventStatusUpdater.updateStatus(event.getId(), ProcessingStatus.RETRY, error);
      rawEventStatusUpdater.incrementRetryCount(event.getId());
    }
  }

Step 2: Create EventDispatcher.java

  public NormalizationResult dispatch(RawEvent event) {
    return switch (event.getEventType()) {
      case "pull_request", "backfill_pull_request" ->
        pullRequestProcessor.process(event);
      case "pull_request_review", "backfill_pull_request_review" ->
        reviewProcessor.process(event);
      case "pull_request_review_comment" ->
        commentProcessor.process(event);
      case "jira:issue_created", "jira:issue_updated", "jira:issue_deleted",
           "backfill_jira_issue" ->
        jiraIssueProcessor.process(event);
      default ->
        NormalizationResult.failure("Unknown event type: " + event.getEventType());
    };
  }

Step 3: Create AiMetadataExtractor.java

  This is the critical component that parses AI usage from PR bodies.

  public AiMetadata extract(String prBody) {
    if (prBody == null || prBody.isEmpty()) {
      return AiMetadata.unknown();
    }

    // Check for explicit non-AI
    boolean hasNoAi = prBody.contains("[x] No AI assistance used")
                   || prBody.contains("[X] No AI assistance used");

    // Check for AI usage categories
    List<String> aiOptions = List.of(
      "AI used for code suggestions",
      "AI used for test generation",
      "AI used for documentation",
      "AI used for refactoring",
      "AI used for debugging or explanation"
    );

    List<String> matchedTypes = new ArrayList<>();
    for (String option : aiOptions) {
      if (prBody.contains("[x] " + option) || prBody.contains("[X] " + option)) {
        matchedTypes.add(option.replace("AI used for ", ""));
      }
    }

    boolean hasAi = !matchedTypes.isEmpty();

    // Check for confidence level
    String confidence = null;
    if (prBody.contains("[x] High") || prBody.contains("[X] High")) confidence = "high";
    else if (prBody.contains("[x] Medium") || prBody.contains("[X] Medium")) confidence = "medium";
    else if (prBody.contains("[x] Low") || prBody.contains("[X] Low")) confidence = "low";

    // Classify
    if (hasNoAi) {
      return AiMetadata.declaredNonAi();
    } else if (hasAi) {
      return AiMetadata.declaredAi(matchedTypes, confidence);
    } else {
      return AiMetadata.unknown();  // ai_assisted = NULL
    }
  }

  Record:
    public record AiMetadata(
      boolean aiAssisted,
      boolean aiUsageInferred,
      String inferenceConfidence,
      List<String> aiUsageTypes,
      String aiAssistanceConfidence
    ) {
      public static AiMetadata declaredNonAi() {
        return new AiMetadata(false, false, null, List.of(), null);
      }
      public static AiMetadata declaredAi(List<String> types, String confidence) {
        return new AiMetadata(true, false, null, types, confidence);
      }
      public static AiMetadata unknown() {
        return new AiMetadata(false, false, null, List.of(), null);
      }
    }

Step 4: Create PullRequestProcessor.java

  public NormalizationResult process(RawEvent event) {
    // 1. Parse payload
    JsonObject payload = JsonParser.parseString(event.getPayload()).getAsJsonObject();

    // 2. Extract fields
    int prNumber = payload.get("number").getAsLong();
    String title = payload.get("title").getAsString();
    String authorLogin = payload.getAsJsonObject("user").get("login").getAsString();
    Instant createdAt = Instant.parse(payload.get("created_at").getAsString());
    Instant mergedAt = payload.has("merged_at") && !payload.get("merged_at").isJsonNull()
      ? Instant.parse(payload.get("merged_at").getAsString()) : null;
    Instant closedAt = payload.has("closed_at") && !payload.get("closed_at").isJsonNull()
      ? Instant.parse(payload.get("closed_at").getAsString()) : null;
    int changedFiles = payload.get("changed_files").getAsInt();
    int additions = payload.get("additions").getAsInt();
    int deletions = payload.get("deletions").getAsInt();
    String body = payload.has("body") ? payload.get("body").getAsString() : "";

    // 3. Look up repository
    String repoName = payload.getAsJsonObject("base").getAsJsonObject("repo").get("name").getAsString();
    String repoOwner = payload.getAsJsonObject("base").getAsJsonObject("repo").getAsJsonObject("owner").get("login").getAsString();
    UUID repositoryId = repositoryRepository.findIdByOwnerAndName(repoOwner, repoName);
    if (repositoryId == null) {
      return NormalizationResult.failure("Repository not found: " + repoOwner + "/" + repoName);
    }

    // 4. Extract AI metadata
    AiMetadata aiMeta = aiMetadataExtractor.extract(body);

    // 5. Pseudonymise author
    String authorId = pseudonymise(authorLogin);

    // 6. Insert or update pull_requests
    PullRequest pr = PullRequest.builder()
      .repositoryId(repositoryId)
      .externalId(String.valueOf(prNumber))
      .title(title)
      .authorId(authorId)
      .createdAtSource(createdAt)
      .mergedAtSource(mergedAt)
      .closedAtSource(closedAt)
      .changedFiles(changedFiles)
      .changedLines(additions + deletions)
      .aiAssisted(aiMeta.aiAssisted())
      .aiUsageInferred(aiMeta.aiUsageInferred())
      .aiUsageInferenceConfidence(aiMeta.inferenceConfidence())
      .aiUsageTypes(aiMeta.aiUsageTypes().toArray(new String[0]))
      .aiAssistanceConfidence(aiMeta.aiAssistanceConfidence())
      .build();

    pullRequestRepository.upsert(pr);  // ON CONFLICT (external_id) DO UPDATE

    return NormalizationResult.success();
  }

  private String pseudonymise(String login) {
    // Simple pseudonymisation: SHA-256 of login + salt
    // Replace with your org's pseudonymisation strategy
    return "dev_" + sha256(login + PSEUDONYM_SALT).substring(0, 12);
  }

Step 5: Create ReviewProcessor.java

  public NormalizationResult process(RawEvent event) {
    JsonObject payload = JsonParser.parseString(event.getPayload()).getAsJsonObject();

    // Extract PR number from pull_request.url field
    String prUrl = payload.getAsJsonObject("pull_request").get("url").getAsString();
    String prNumber = prUrl.substring(prUrl.lastIndexOf("/") + 1);

    // Look up pull_request_id
    UUID prId = pullRequestRepository.findIdByExternalId(prNumber);
    if (prId == null) return NormalizationResult.failure("PR not found: " + prNumber);

    // Extract review data
    String reviewerLogin = payload.getAsJsonObject("user").get("login").getAsString();
    String state = payload.get("state").getAsString();  // APPROVED, CHANGES_REQUESTED, COMMENTED
    Instant submittedAt = Instant.parse(payload.get("submitted_at").getAsString());

    // Update or insert review_analytics
    // For first_review_at: use MIN(submitted_at) across all reviews for this PR
    // For approved_at: use submitted_at where state = "APPROVED"
    reviewAnalyticsRepository.upsertFirstReviewIfEarlier(prId, submittedAt);
    if ("APPROVED".equals(state)) {
      reviewAnalyticsRepository.updateApprovedAt(prId, submittedAt);
    }
    reviewAnalyticsRepository.incrementCommentsCount(prId);

    return NormalizationResult.success();
  }

Step 6: Create JiraIssueProcessor.java

  public NormalizationResult process(RawEvent event) {
    JsonObject payload = JsonParser.parseString(event.getPayload()).getAsJsonObject();
    JsonObject issue = payload.getAsJsonObject("issue");

    String issueKey = issue.get("key").getAsString();
    String issueType = issue.getAsJsonObject("fields").getAsJsonObject("issuetype").get("name").getAsString();
    String severity = issue.getAsJsonObject("fields").getAsJsonObject("priority").get("name").getAsString();
    String status = issue.getAsJsonObject("fields").getAsJsonObject("status").get("name").getAsString();
    Instant createdAt = Instant.parse(issue.getAsJsonObject("fields").get("created").getAsString());
    String resolvedAtStr = issue.getAsJsonObject("fields").get("resolutiondate").isJsonNull()
      ? null : issue.getAsJsonObject("fields").get("resolutiondate").getAsString();
    Instant resolvedAt = resolvedAtStr != null ? Instant.parse(resolvedAtStr) : null;

    // Try to link to a PR via issue links
    UUID linkedPrId = jiraLinker.findLinkedPrId(issue);

    JiraIssue jiraIssue = JiraIssue.builder()
      .externalKey(issueKey)
      .issueType(issueType)
      .severity(severity)
      .status(status)
      .linkedPullRequestId(linkedPrId)
      .createdAtSource(createdAt)
      .ResolvedAtSource(resolvedAt)
      .build();

    jiraIssueRepository.upsert(jiraIssue);  // ON CONFLICT (external_key) DO UPDATE

    return NormalizationResult.success();
  }

Step 7: Create JiraLinker.java

  public UUID findLinkedPrId(JsonObject issue) {
    // Strategy 1: Check issue links for "is developed by" type
    JsonArray issueLinks = issue.getAsJsonObject("fields").getAsJsonArray("issuelinks");
    if (issueLinks != null) {
      for (JsonElement link : issueLinks) {
        JsonObject linkObj = link.getAsJsonObject();
        if ("is developed by".equals(linkObj.getAsJsonObject("type").get("inward").getAsString())) {
          String linkedPrKey = linkObj.getAsJsonObject("outwardIssue").get("key").getAsString();
          // This is the Jira key, not the PR number. Need to search commit messages.
        }
      }
    }

    // Strategy 2: Search for Jira key in commit messages
    // GET /repos/{owner}/{repo}/commits?path=&sha=&per_page=100
    // Check commit messages for pattern: PILOT-123

    // Strategy 3: Search for Jira key in branch names
    // GET /repos/{owner}/{repo}/branches?per_page=100
    // Check branch names for pattern: feature/PILOT-123-*

    // Return the PR ID if found, null otherwise.
    return null;  // implement based on your org's conventions
  }
